\section{Individual Responsibilities}

\begin{frame}\frametitle{Individual Responsibilities}
\begin{itemize}
    \item \textbf{Architecture/Integration (Role):} plugin skeleton, module boundaries, build integration, and validation/QA ownership.
    \item \textbf{Audio Engine (Role):} voice management, routing, and end-to-end realtime-safe orchestration.
    \item \textbf{DSP / Audio Effects (Role):} DSP building blocks (e.g., mixer, filters), algorithm implementation and tuning.
    \item \textbf{UI + State Handoff (Role):} editor controls, gain visualization, and parameter linking.
\vspace{0.5ex}

\end{itemize}
\end{frame}
