\section{Goals and Scope}

\begin{frame}\frametitle{Goals and Scope}
\begin{itemize} [<+->]
    \item \textbf{Goal 1:} Core DSP blocks for volumetric synthesis.
    \item \textbf{Goal 2:} Playable plugin from MIDI to audio out.
    \item \textbf{Goal 3:} Simple UI for 3D mixing.
    \item \textbf{In scope:} Oscillator, filter, envelope, mixer, wiring, tests, UI state.
    \item \textbf{Out of scope:} Advanced FX, many presets, visual polish.
\end{itemize}
\end{frame}
