\section{Goals and Scope}

\begin{frame}\frametitle{Goals and Scope}
\begin{itemize}
    \item \textbf{Goal 1:} Build stable core DSP blocks for a volumetric synthesis pipeline.
    \item \textbf{Goal 2:} Integrate modules into a playable plugin path (MIDI to audio out).
    \item \textbf{Goal 3:} Provide a clear UI/control layer for 3D mix behavior.
    \item \textbf{In scope:} oscillator/filter/envelope/mixer integration, plugin wiring, tests, and core UI state sharing.
    \item \textbf{Out of scope:} advanced FX suite, large preset ecosystem, and deep visual polish.
\end{itemize}
\end{frame}
